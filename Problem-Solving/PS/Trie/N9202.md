private static void DFS(int y, int x, String word, int cnt){

    if(cnt > 8)return;

    if(!trie.check(word)){
        return;
    };

    for(int i=0;i<8;i++){
        int ny = y + dy[i];
        int nx = x + dx[i];

        if(ny >= 4 || nx >= 4 || ny < 0 || nx < 0)continue;
        if(check[ny][nx])continue;

        word += board[ny][nx];
        check[ny][nx] = true;
        DFS(ny, nx, word, cnt+1);
        word = word.substring(0, word.length()-1);
        check[ny][nx] = false;
    }
}
